## âš¡ STEP 1: CREATE PROJECT & INSTALL DEPENDENCIES

```bash
# 1. Create React Native Project
npx react-native@latest init ContactSMSApp
cd ContactSMSApp

# 2. Install ALL Dependencies
npm install @react-native-async-storage/async-storage@1.21.0
npm install @react-native-community/geolocation@3.2.1
npm install @react-native-firebase/app@19.0.1
npm install @react-native-firebase/storage@19.0.1
npm install @react-navigation/native@6.1.9
npm install @react-navigation/native-stack@6.9.17
npm install react-native-audio-recorder-player@3.6.0
npm install react-native-fs@2.20.0
npm install react-native-get-sms-android@1.0.5
npm install react-native-permissions@4.0.3
npm install react-native-safe-area-context@4.8.2
npm install react-native-screens@3.29.0
npm install react-native-share@10.0.2
npm install react-native-vector-icons@10.0.3
npm install react-native-dotenv@3.4.9

# 3. Create Folder Structure
mkdir -p src/context src/components src/screens src/services src/utils src/navigation

# 4. Link native modules
cd android && ./gradlew clean && cd ..
```
